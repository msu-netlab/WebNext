    $Domains = DomainsWithIP(ip)$

        $seq = (domain_i, domain_{i+1},..., domain_S)$
        
        \If{$seq \in H$}{
            $H[seq] = H[seq] + 1$
        }
        \Else{
            $H[seq] = 1$
        }
    }
